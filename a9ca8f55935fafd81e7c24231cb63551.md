# date-logic


```js
var calendar = new DateLogic({
    onChange: function (data) {
        // data format equal monthData() reutrn
        self.setState({
            calendar: data
        })
    }
})
var data = calendar.getData()
// data format equal monthData() reutrn

calendar.change('2017', '9', '11')
// call onChange
calendar.lastMonth()
// call onChange
calendar.nextMonth()
// call onChange
calendar.lastYear()
// call onChange
calendar.nextYear()
// call onChange
```
